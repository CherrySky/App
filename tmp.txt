if (first == second ||
                Math.abs(first.length() - second.length()) >= 2) {
            return false;
        }

        int firstIndex = 0;
        int secondIndex = 0;

        boolean isOneEdit = false;
        while (firstIndex < first.length()) {
            if (first.charAt(firstIndex) != second.charAt(secondIndex)) {
                if (isOneEdit) return false;
                isOneEdit = true;
                if (first.length() > second.length()) {
                    firstIndex++;
                } else if (second.length() > first.length()) {
                    secondIndex++;
                } else {
                    firstIndex++;
                    secondIndex++;
                }
            } else {
                firstIndex++;
                secondIndex++;
            }
        }
        return isOneEdit;
